-- =====================================================================
-- Seed demo: 5 member yang baru MEMPERPANJANG membership.
-- Tiap renewal memperpanjang masa berlaku sesuai durasi paket member, dan
-- kolom membership_expiry_date member ikut diperbarui agar konsisten.
-- Aman dijalankan sekali (menambah 5 baris ke member_renewals).
-- =====================================================================

USE `pos_gym`;

-- Pola per member: catat renewal (prev = expiry sekarang, new = + durasi paket,
-- amount = harga paket), lalu perpanjang masa berlaku member.

-- 1) Aditya Pratama (PKG-001) - perpanjang hari ini.
INSERT INTO `member_renewals`
  (member_code, package_code, previous_expiry_date, new_expiry_date, amount, renewed_at)
SELECT m.member_code, m.package_code, m.membership_expiry_date,
       DATE_ADD(m.membership_expiry_date, INTERVAL g.duration_days DAY), g.price, '2026-06-28'
FROM members m JOIN gym_packages g ON g.package_code = m.package_code
WHERE m.member_code = 'MBR-96018192';
UPDATE members m JOIN gym_packages g ON g.package_code = m.package_code
SET m.membership_expiry_date = DATE_ADD(m.membership_expiry_date, INTERVAL g.duration_days DAY)
WHERE m.member_code = 'MBR-96018192';
